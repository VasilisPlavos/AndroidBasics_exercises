sqlite3 <filename> = έτσι αρχίζουν όλα
.open <filename> = ανοίγει ένα αρχείο με schemas. Αν το αρχείο δεν υπάρχει το δημιουργεί
.tables = Βλέπεις τα tables που έχει μέσα το αρχείο σου

Σχετικά με το sqlite3 <filename> και το .open <filename>:
Αν είσαι σε περιβάλλον commander και όχι μέσα στην sqlite3 τότε ξεκινάς κάπως έτσι:
C:\>sqlite3 <filename>

Αλλιώς ξεκινάς κάπως έτσι:
sqlite>.open <filename>


Φτιάχνεις ένα πίνακα που θέλεις:
CREATE TABLE pets(_id INTEGER, name TEXT, breed TEXT, gender INTEGER, weight INTEGER);

DROP TABLE pets = σβήνει τον πίνακα


Προσθέτει μερικά στοιχεία:
INSERT INTO pets(_id, name, breed, gender, weight) VALUES(1, "rex", "unknown", 0, 12);
INSERT INTO pets(_id, name, breed, gender, weight) VALUES(1, "rex", "unknown", 0, 12);
INSERT INTO pets(_id, name, breed, gender, weight) VALUES(2, "garf", "tabby", 1, 8);
Σημείωση: Στο παραπάνω δεν έχω κάνει μοναδικό το id άρα μπορούν να προκύψουν διπλότυπα ids


SELECT * FROM pets; Βλέπεις τι έχει μέσα ο πίνακας pets
SELECT * FROM pets WHERE _id == 1; Βλέπεις στον πίνακα pets όποιο κατοικίδιο έχει _id = 1
SELECT * FROM pets WHERE _id != 1; Βλέπεις στον πίνακα pets όποιο κατοικίδιο έχει _id διαφορετικό από 1
SELECT name, breed FROM pets; Εκτυπώνει μόνο αυτές τις 2 στήλες
SELECT name, breed FROM pets WHERE _id==2;
SELECT * FROM pets ORDER BY name ASC/DESC; Για να sortάρεις την λίστα

PRIMARY KEY = Μοναδικό
AUTOINCREMENT = Κάθε φορά αυξάνεται +1
NOT NULL = Δεν μπορεί να είναι άδειο το πεδίο στην εισαγωγή
DEFAULT 0 = Το πεδίο θα είναι μηδέν αν δε δώσουμε κάποιο νούμερο

Παράδειγμα:
CREATE TABLE pets2(_id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT NOT NULL, breed TEXT, gender INTEGER NOT NULL, weight INTEGER NOT NULL DEFAULT 0);
INSERT INTO pets2(name, breed, gender, weight) VALUES("rex", "tabby", 1, 8);
INSERT INTO pets2(name, breed, gender) VALUES("rex", "tabby", 1);

.header on = εμφανίζει και το header του πίνακα
.mode column = στα δείχνει σε δομή πίνακα, βοηθάει να τα καταλάβει ευκολότερα το ανθρώπινο μάτι
